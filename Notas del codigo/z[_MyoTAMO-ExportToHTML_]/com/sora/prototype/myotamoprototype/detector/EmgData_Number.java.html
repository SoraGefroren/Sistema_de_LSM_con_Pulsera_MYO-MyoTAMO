39           ByteReader emgData = new ByteReader(); 
40           emgData.setByteData(byteData); 
41           // Formar los datos EMG de 16 elementos 
42           for (int i_emg16 = 0; i_emg16 < 16; i_emg16++) { 
43               emgData_tam16[i_emg16] = emgData.getByte(); 
44           } 
45           // Formar los datos EMG de 08 elementos 
46           for (int i_emg8 = 0; i_emg8 < 8; i_emg8++) { 
47               if (Math.abs(emgData_tam16[i_emg8]) < Math.abs(emgData_tam16[i_emg8 + 8])){ 
48                   emgData_tam08[i_emg8] = Math.abs(emgData_tam16[i_emg8 + 8]); 
49               } else { 
50                   emgData_tam08[i_emg8] = Math.abs(emgData_tam16[i_emg8]); 
51               } 
52           } 
53       }
