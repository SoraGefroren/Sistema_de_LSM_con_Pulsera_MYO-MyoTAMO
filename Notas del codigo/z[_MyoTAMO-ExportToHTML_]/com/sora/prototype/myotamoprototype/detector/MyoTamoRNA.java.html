3    // Libreria 
4    import android.util.Log; 
5    import java.util.ArrayList;

7    /** 
8     * Implementacion del comportamiento de una RNA sin retroalimentacion 
9     */ 
10   public class MyoTamoRNA { 
11       // Variable de Pesos para esta RNA 
12       private ArrayList<Double[][]> weightsRNA = new ArrayList<Double[][]>(); 
13       // Variable de Bias para esta RNA 
14       private ArrayList<Double[]> biasRNA = new ArrayList<Double[]>();; 
15       // Variable de Funciones para esta RNA 
16       private String[] nameFuctionsForLayer = {""}; 
17       // Variable de status (Estado) de esta RNA 
18       private boolean statusRNA = false; 
19       // TAG para la Visualizacion de MSJ en Consola 
20       private static final String TAG = "MyoTAMO-RNA"; 
21       // ---------------------------------------------- ---------------------------------------------- 
22       // Contruccion  -------------------------------------------------------------------------------- 
23       /** 
24        * Constructor de la implementacion de la RNA 
25        */ 
26       public MyoTamoRNA(ArrayList<Double[][]> weights, ArrayList<Double[]> bias, String[] nomFuctionsForLayer){ 
27           weightsRNA = weights; // Establecer los Pesos 
28           biasRNA = bias;       // Establecer las Bias 
29           nameFuctionsForLayer = nomFuctionsForLayer; // Establecer las funciones por capa 
30           if((weightsRNA.size() == nameFuctionsForLayer.length) && (weightsRNA.size() == biasRNA.size())){ 
31               statusRNA = true; // Correcta contruccion de la RNA 
32           } 
33       } 
34       /** 
35        * Constructor vacio de la implemementacion de la RNA 
36        */ 
37       public MyoTamoRNA(){ } // Incorrecta contruccion de la RNA 
38       // ---------------------------------------------- ---------------------------------------------- 
39       // Calculo RNA --------------------------------------------------------------------------------- 
40       /** 
41        * En base a un patron de una muestra de datos EMG, devuelve un patron numerico de 0 y 1, el cual podria corresponder con una seña 
42        * @param averageEmgData Patron numerico de muestras de datos EMG 
43        * @param seeEval Indicacion para ver o no mensajes en consola 
44        * @return Patron numerico de 0 y 1 que podria coincidir con el de una seña 
45        */ 
46       public String getStringRNAResult_ofEmgData(Double[] averageEmgData, boolean seeEval){ 
47           String rnaResult = ""; // Resultado del analisis por la RNA 
48           if(statusRNA){ // Comprobar el estado de la RNA 
49               Double[] inputs = averageEmgData; // Entradas 
50               // Ciclo para recorrer cada capa de la RNA 
51               for(int numLayer = 0; numLayer < weightsRNA.size(); numLayer++){ 
52                   Double[][] weights = weightsRNA.get(numLayer); // Pesos de la presente capa 
53                   Double[] bias = biasRNA.get(numLayer);         // Bias de la presente capa 
54                   int numNeurons = weights.length; // Numero de Neuronas en las presente capa 
55                   int numInputs = inputs.length;   // Numero de entradas para la presente capa 
56                   Double[] resultOnNeurons = new Double[numNeurons]; // Resultado de la presente capa 
57                   if(numNeurons == bias.length){ 
58                       // Operacion por cada Neurona 
59                       for (int nNeuron = 0; nNeuron < numNeurons; nNeuron++){ 
60                           resultOnNeurons[nNeuron] = 0.0; // Resultado por defecto de la Neurona 
61                           if(numInputs == weights[nNeuron].length){ 
62                               // Operacion por cada Entrada 
63                               for(int nInput = 0; nInput < numInputs; nInput++){ 
64                                   // Resultado de la Neurona 
65                                   resultOnNeurons[nNeuron] += (inputs[nInput] * weights[nNeuron][nInput]); 
66                               } 
67                               resultOnNeurons[nNeuron] += bias[nNeuron]; // Sumar Bias al resultado de la Neurona 
68                               // Aplicar funcion especifica al resultado de la Neurona 
69                               resultOnNeurons[nNeuron] = getResult_activationFunction(nameFuctionsForLayer[numLayer], resultOnNeurons[nNeuron]); 
70                           }else{ 
71                               if(seeEval){ 
72                                   Log.d(TAG, "------ N°Inputs:[" + numInputs + "], N°wInpt: [" + weights[nNeuron].length + "]"); 
73                               } 
74                           } 
75                       } 
76                   }else{ 
77                       if(seeEval){ 
78                           Log.d(TAG, "------ N°Neurons:[" + numNeurons + "], N°Bias: [" + bias.length + "]"); 
79                       } 
80                   } 
81                   // Actualizar la entrada con el ultimo resultado de las Neuronas 
82                   inputs = resultOnNeurons; 
83               } 
84               // Convertir el resultado de la RNA a un patron numerico 
85               for(int i = 0; i < inputs.length; i++){ 
86                   if(inputs[i] > 0.9){ 
87                       rnaResult += "1"; 
88                   }else{ 
89                       rnaResult += "0"; 
90                   } 
91                   if(seeEval){ 
92                       Log.d(TAG, "------ Nr°:"+i+"...[" + inputs[i] + "]["+rnaResult+"]"); 
93                   } 
94               } 
95           }else{ 
96               if(seeEval){ 
97                   Log.d(TAG, ">>>>>> myoTamoRNA Uninitialized <<<<<<"); 
98               } 
99           } 
100          return rnaResult; 
101      } 
102      // Aplicar una funcion especifica a un valor especifico 
103      /** 
104       * Devuelve la aplicacion de una funcion especifica sobre un valor especifico 
105       * @param nomFunction Nombre de la funcion a ser aplicada 
106       * @param value Valor sobre el cual se aplicara una funcion 
107       * @return Resultado de la aplicacion una funcion sobre un valor 
108       */ 
109      public Double getResult_activationFunction(String nomFunction, Double value){ 
110          Double result; // Resultado de la funcion 
111          switch (nomFunction){ 
112              case "logsig": // Aplicar funcion logaritmica sigmoide 
113                  result = (1 / (1 + Math.exp(-value))); 
114                  break; 
115              case "tansig": // Aplicar funcion de tangente sigmoide 
116                  result = (2 / (1+Math.exp(-2 * value))) -1; 
117                  break; 
118              case "purelin": // Funcion de tranferencia linear 
119                  result = value; 
120                  break; 
121              default: 
122                  result = value; 
123          } 
124          return result; 
125      }
