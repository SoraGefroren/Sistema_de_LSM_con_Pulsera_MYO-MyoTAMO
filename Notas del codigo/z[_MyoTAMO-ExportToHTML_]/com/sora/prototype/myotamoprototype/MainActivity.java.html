94       @Override // Cada vez que se escoga una opción del menu de opciones 
95       public boolean onOptionsItemSelected(MenuItem item) { 
96           switch (item.getItemId()) { 
97               case MENU_CONECT: // Cargar Actividad para conexión Bluetooth 
98                   Intent intentListBluetoothActivity = new Intent(this,ListBluetoothActivity.class); 
99                   startActivity(intentListBluetoothActivity); 
100                  return true; 
101              case BTN_DESCONECT: // Terminar conexión Bluetooth 
102                  // Emitir mensaje en pantalla 
103                  Toast.makeText(getApplicationContext(), "Pulsera MYO desconectada", Toast.LENGTH_SHORT).show(); 
104                  closeBLEGatt(); 
105                  return true; 
106              case BTN_TEST_CONECT: // Probar conexión con Pulsera MYO (Hacer vibrar a la Pulsera) 
107                  if (mBluetoothGatt == null || !mMyoCallback.setMyoControlCommand(commandList.sendVibration3())) { 
108                      // Emitir mensaje en pantalla 
109                      Toast.makeText(getApplicationContext(), "No hay conexión con la Pulsera MYO", Toast.LENGTH_SHORT).show(); 
110                  } 
111                  return true; 
112          } 
113          return false; 
114      }

124      @Override // Cuando se busca la conexion Bluetooth con una Pulsera 
125      public void onLeScan(BluetoothDevice device, int rssi, byte[] scanRecord) { 
126          if (deviceName.equals(device.getName())) { 
127              mBluetoothAdapter.stopLeScan(this); 
128              mMyoCallback = new MyoGattCallback(mHandler, txt_textZone, txt_gestureZone, txt_rnaTextZone, linesGraph, txt_sensorsEmgData); 
129              mBluetoothGatt = device.connectGatt(this, false, mMyoCallback); 
130              mMyoCallback.setBluetoothGatt(mBluetoothGatt); 
131          } 
132      }

136          if (requestCode == REQUEST_ENABLE_BT && resultCode == RESULT_OK){

140                      mBluetoothAdapter.stopLeScan(MainActivity.this);

219      // Conectar Dispositivo MYO (Iniciar conexión bluetooth) 
220      /** 
221       * Para conectar la Pulsera MYO 
222       */ 
223      public void conectMYODevice(){ 
224          if (deviceName != null && !isConnected) { 
225              isConnected = true; 
226              // Ensures Bluetooth is available on the device and it is enabled. If not, 
227              // displays a dialog requesting user permission to enable Bluetooth. 
228              if (mBluetoothAdapter == null || !mBluetoothAdapter.isEnabled()) { 
229                  Intent enableBtIntent = new Intent(BluetoothAdapter.ACTION_REQUEST_ENABLE); 
230                  startActivityForResult(enableBtIntent, REQUEST_ENABLE_BT); 
231              } else { 
232                  // Scanning Time out by Handler. 
233                  // The device scanning needs high energy. 
234                  mHandler.postDelayed(new Runnable() { 
235                      @Override 
236                      public void run() { 
237                          mBluetoothAdapter.stopLeScan(MainActivity.this); 
238                      } 
239                  }, SCAN_PERIOD); 
240                  mBluetoothAdapter.startLeScan(this); 
241              } 
242          } 
243      } 
244      // Click en Botón, para Iniciar/Detener el proceso de detección de información EMG 
245      /** 
246       * Metodo para el boton de "INICIAR" o "DETENER" la deteccion de datos EMG 
247       */ 
248      public void onClickEMGDetect(View v) { 
249          if(!doDetecting){ // Si no esta detectando datos EMG, entonces se Inicia esta detección 
250              if (mBluetoothGatt == null || !mMyoCallback.setMyoControlCommand(commandList.sendEmgOnly())) { 
251                  Log.d(TAG, "No se puede iniciar la detección de señas"); 
252              } 
253              // Emitir mensaje en pantalla 
254              Toast.makeText(getApplicationContext(), "Detección de señas iniciada ;D", Toast.LENGTH_SHORT).show(); 
255              btn_detectEMG.setText("Detener"); 
256              doDetecting = true; 
257              cleanTextZone(); 
258          }else{ // Si esta detectando datos EMG, entonces se detiene esta detección 
259              if (mBluetoothGatt == null 
260                      || !mMyoCallback.setMyoControlCommand(commandList.sendUnsetData()) 
261                      || !mMyoCallback.setMyoControlCommand(commandList.sendNormalSleep())) { 
262                  Log.d(TAG, "No se puede detener la detección de señas"); 
263              } 
264              // Emitir mensaje en pantalla 
265              Toast.makeText(getApplicationContext(), "Detección de señas detenida :|", Toast.LENGTH_SHORT).show(); 
266              btn_detectEMG.setText("Iniciar"); 
267              doDetecting = false; 
268          } 
269      }
