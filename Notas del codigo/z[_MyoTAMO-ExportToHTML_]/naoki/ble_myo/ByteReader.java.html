1    package naoki.ble_myo; 
2     
3    import java.nio.ByteBuffer; 
4    import java.nio.ByteOrder; 
5     
6    /** 
7     * Created by naoki on 15/04/06. 
8     *  
9     * This class help you to read the byte line from Myo. 
10    * But be carefully to byte array size. There is no limitation of get() method,  
11    * so there is a possibilty of overloading the byte buffer. 
12    *  
13    */ 
14    
15   public class ByteReader { 
16       private byte[] byteData; 
17       private ByteBuffer bbf; 
18    
19       public void setByteData(byte[] data){ 
20           this.byteData = data; 
21           this.bbf = ByteBuffer.wrap(this.byteData); 
22           bbf.order(ByteOrder.LITTLE_ENDIAN); 
23       } 
24    
25       public byte[] getByteData() { 
26           return byteData; 
27       } 
28    
29       public short getShort() { 
30           return this.bbf.getShort(); 
31       } 
32    
33       public byte getByte(){ 
34           return this.bbf.get(); 
35       } 
36    
37       public int getInt(){ 
38           return this.bbf.getInt(); 
39       } 
40    
41       public String getByteDataString() { 
42           final StringBuilder stringBuilder = new StringBuilder(byteData.length); 
43           for (byte byteChar : byteData) { 
44               stringBuilder.append(String.format("%02X ", byteChar)); 
45           } 
46           return stringBuilder.toString(); 
47       } 
48    
49       public String getIntDataString() { 
50           final StringBuilder stringBuilder = new StringBuilder(byteData.length); 
51           for (byte byteChar : byteData) { 
52               stringBuilder.append(String.format("%5d,", byteChar)); 
53           } 
54           return stringBuilder.toString(); 
55       } 
56   }
