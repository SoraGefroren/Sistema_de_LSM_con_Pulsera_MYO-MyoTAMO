94       @Override 
95       public void onConnectionStateChange(BluetoothGatt gatt, int status, int newState) { 
96           super.onConnectionStateChange(gatt, status, newState); 
97           Log.d(TAG, "La conexión cambio de estado: " + status + " -> " + newState); 
98           if (newState == BluetoothProfile.STATE_CONNECTED) { 
99               // GATT Connected // Searching GATT Service 
100              gatt.discoverServices(); 
101          } else if (newState == BluetoothProfile.STATE_DISCONNECTED) { 
102              // GATT Disconnected 
103              stopCallback(); 
104              Log.d(TAG,"Bluetooth Desconectado"); 
105          } 
106      } 
107      @Override 
108      public void onServicesDiscovered(BluetoothGatt gatt, int status) { 
109          super.onServicesDiscovered(gatt, status); 
110          Log.d(TAG, "onServicesDiscovered received: " + status); 
111          if (status == BluetoothGatt.GATT_SUCCESS) { 
112              // Find GATT Service 
113              BluetoothGattService service_emg = gatt.getService(UUID.fromString(MYO_EMG_DATA_ID)); 
114              if (service_emg == null) { 
115                  Log.d(TAG,"No Myo EMG-Data Service !!"); 
116              } else { 
117                  Log.d(TAG, "Find Myo EMG-Data Service !!"); 
118                  // Getting CommandCharacteristic 
119                  mCharacteristic_emg0 = service_emg.getCharacteristic(UUID.fromString(EMG_0_ID)); 
120                  if (mCharacteristic_emg0 == null) { 
121                      callback_msg = "No se encontro EMG-Data"; 
122                  } else { 
123                      // Setting the notification 
124                      boolean registered_0 = gatt.setCharacteristicNotification(mCharacteristic_emg0, true); 
125                      if (!registered_0) { 
126                          Log.d(TAG,"EMG-Data Notification FALSE !!"); 
127                      } else { 
128                          Log.d(TAG,"EMG-Data Notification TRUE !!"); 
129                          // Turn ON the Characteristic Notification 
130                          BluetoothGattDescriptor descriptor_0 = mCharacteristic_emg0.getDescriptor(UUID.fromString(CLIENT_CHARACTERISTIC_CONFIG)); 
131                          if (descriptor_0 != null ){ 
132                              descriptor_0.setValue(BluetoothGattDescriptor.ENABLE_NOTIFICATION_VALUE); 
133                              writeGattDescriptor(descriptor_0); 
134                              Log.d(TAG,"Set descriptor"); 
135                          } else { 
136                              Log.d(TAG,"No descriptor"); 
137                          } 
138                      } 
139                  } 
140              } 
141              BluetoothGattService service = gatt.getService(UUID.fromString(MYO_CONTROL_ID)); 
142              if (service == null) { 
143                  Log.d(TAG,"No Myo Control Service !!"); 
144              } else { 
145                  Log.d(TAG, "Find Myo Control Service !!"); 
146                  // Get the MyoInfoCharacteristic 
147                  BluetoothGattCharacteristic characteristic = 
148                          service.getCharacteristic(UUID.fromString(MYO_INFO_ID)); 
149                  if (characteristic == null) { 
150                  } else { 
151                      Log.d(TAG, "Find read Characteristic !!"); 
152                      //put the characteristic into the read queue 
153                      readCharacteristicQueue.add(characteristic); 
154                      //if there is only 1 item in the queue, then read it.  If more than 1, we handle asynchronously in the callback above 
155                      //GIVE PRECEDENCE to descriptor writes.  They must all finish first. 
156                      if((readCharacteristicQueue.size() == 1) && (descriptorWriteQueue.size() == 0)) { 
157                          mBluetoothGatt.readCharacteristic(characteristic); 
158                      } 
159                  } 
160                  // Get CommandCharacteristic 
161                  mCharacteristic_command = service.getCharacteristic(UUID.fromString(COMMAND_ID)); 
162                  if (mCharacteristic_command == null) { 
163                  } else { 
164                      Log.d(TAG, "Find command Characteristic !!"); 
165                  } 
166              } 
167          }

169      @Override 
170      public void onDescriptorWrite(BluetoothGatt gatt, BluetoothGattDescriptor descriptor, int status) { 
171          if (status == BluetoothGatt.GATT_SUCCESS) { 
172              Log.d(TAG, "Callback: Wrote GATT Descriptor successfully."); 
173          } 
174          else{ 
175              Log.d(TAG, "Callback: Error writing GATT Descriptor: "+ status); 
176          } 
177          descriptorWriteQueue.remove();  //pop the item that we just finishing writing 
178          //if there is more to write, do it! 
179          if(descriptorWriteQueue.size() > 0) 
180              mBluetoothGatt.writeDescriptor(descriptorWriteQueue.element()); 
181          else if(readCharacteristicQueue.size() > 0) 
182              mBluetoothGatt.readCharacteristic(readCharacteristicQueue.element()); 
183      }

435       * Este metodo indica cuando se ha establecido conexion con la Pulsera MYO 
436       */ 
437      public void onCharacteristicRead(BluetoothGatt gatt, BluetoothGattCharacteristic characteristic, int status) { 
438          readCharacteristicQueue.remove(); 
439          if (status == BluetoothGatt.GATT_SUCCESS) { 
440              if (UUID.fromString(FIRMWARE_ID).equals(characteristic.getUuid())) { 
441                  // Myo Firmware Infomation 
442                  final byte[] data = characteristic.getValue(); 
443                  if (data != null && data.length > 0) { 
444                      ByteReader byteReader = new ByteReader(); 
445                      byteReader.setByteData(data); 
446                      Log.d(TAG, String.format("This Version is %d.%d.%d - %d", 
447                              byteReader.getShort(), byteReader.getShort(), 
448                              byteReader.getShort(), byteReader.getShort())); 
449                  } 
450                  if (data == null) { Log.d(TAG,"Characteristic String is " + characteristic.toString()); } 
451              } else if (UUID.fromString(MYO_INFO_ID).equals(characteristic.getUuid())) { 
452                  // Myo Device Information 
453                  final byte[] data = characteristic.getValue(); 
454                  if (data != null && data.length > 0) { 
455                      // ----------------------------------------------------------------------------- 
456                      // ----------------------------------------------------------------------------- 
457                      /* Para indicar que ya se puede iniciar la deteccion de señas, 
458                         Nos dice que se ha establecido conexion con una Pulsera MYO */ 
459                      callback_msg = "Ya puedes iniciar la detección de señas."; 
460                      mHandler.post(new Runnable() { 
461                          @Override 
462                          public void run() { 
463                              txt_textZone.setText(callback_msg); 
464                          } 
465                      }); 
466                      // ----------------------------------------------------------------------------- 
467                      // ----------------------------------------------------------------------------- 
468                  } 
469              } 
470          } else { Log.d(TAG, "onCharacteristicRead error: " + status); } 
471          if(readCharacteristicQueue.size() > 0){ mBluetoothGatt.readCharacteristic(readCharacteristicQueue.element()); } 
472      }

478      public boolean setMyoControlCommand(byte[] command) { 
479          if ( mCharacteristic_command != null) { 
480              mCharacteristic_command.setValue(command); 
481              int i_prop = mCharacteristic_command.getProperties(); 
482              if (i_prop == BluetoothGattCharacteristic.PROPERTY_WRITE) { 
483                  if (mBluetoothGatt.writeCharacteristic(mCharacteristic_command)) { 
484                      return true; 
485                  } 
486              } 
487          } 
488          return false; 
489      }

495      public void stopCallback() { 
496          // Before the closing GATT, set Myo [Normal Sleep Mode]. 
497          setMyoControlCommand(commandList.sendNormalSleep()); 
498          descriptorWriteQueue = new LinkedList<BluetoothGattDescriptor>(); 
499          readCharacteristicQueue = new LinkedList<BluetoothGattCharacteristic>(); 
500          if (mCharacteristic_command != null) { 
501              mCharacteristic_command = null; 
502          } 
503          if (mCharacteristic_emg0 != null) { 
504              mCharacteristic_emg0 = null; 
505          } 
506          if (mBluetoothGatt != null) { 
507              mBluetoothGatt = null; 
508          }
