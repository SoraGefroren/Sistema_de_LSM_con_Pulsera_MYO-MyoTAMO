56       /** 
57        * Este metodo analisa la cadena de caracteres, buscando la ultima palabra la cual analizara para intentar predecir completamente en base con un diccionario de palabras 
58        * @param message Mensaje o Palabra que sera analizada 
59        * @return Mensaje con la ultima palabra compuesta (es decir su prediccion) 
60        */ 
61       public String fixedUltimateWord(String message){ 
62           String returnPhrase = ""; // Resultado o frase 
63           if(message != null && message != ""){ // Comprobar el tipo de mensaje de entrada 
64               // Si solo se tiene 1 palabra, devolver el tratamiento de esta 
65               if( message.split(" ").length < 1 ){ 
66                   returnPhrase = fixedWord(message); 
67               }else{ // Si se tiene mas de 1 palabra, devolver el tratamiento de la ultima palabra 
68                   String[] wordsInMessage = message.split(" "); // Separar el texto por espacios 
69                   // Respaldar todas las palabras excepto la ultima 
70                   for(int i = 0; i < wordsInMessage.length -1; i++){ 
71                       returnPhrase += wordsInMessage[i] + " "; 
72                   } 
73                   // Concatenar al resultado el tratamiento de la ultima palabra 
74                   returnPhrase += fixedWord(wordsInMessage[wordsInMessage.length - 1]); 
75               } 
76           }else{ 
77               returnPhrase = message; // Devolver el mismo mensaje de entrada 
78           } 
79           return returnPhrase; 
80       } 
81       // Tratar una plabra en particular 
82       /** 
83        * Analisa una palabra en particular para intentar completarla con base al diccionario de palabras 
84        * @param wordString Palabra de entrada que sera analisada (esta secuencia de caracteres podria no ser alguna palabra en especifico) 
85        * @return La palabra de entrada completada en base al analisis con el diccionario de palabras 
86        */ 
87       private int minMatch = 2;              // Minimo numero de caracteres para considerar al texto como palabra 
88       private int minMatchSize = minMatch*2; // Minimo numero de caracteres para comenzar a tratar una palabra 
89       public String fixedWord(String wordString){ 
90           String newWord = wordString;                     // Resultado del tratamiento de la palabra, por defecto el mismo texto de entrada 
91           if(wordString.toCharArray().length >= minMatch){ // Verificar si se puede conciderar palabra 
92               boolean breakFixed = false; // Variable de control para romper comparacion con el diccionario en caso de encontrar coincidencia 
93               for(ArrayList<String> arrayWords :arrayOfWords){ // Recorrer los conjunto de palabras del diccionario 
94                   // Comparar cada palabra del diccionario con la palabra de entrada 
95                   for(String wordOfBD: arrayWords){ // Recorrer cada palabra de un conjunto de palabras del diccionario 
96                       Log.d(TAG, "__________________."+wordOfBD+".vs."+wordString+".__________________"); 
97                       String myMiniumWord = ""; 
98                       // Limpiar la Palabra del diccionario de impuresas como espacios 
99                       if(wordOfBD.toCharArray().length >= minMatchSize){ 
100                          myMiniumWord = wordOfBD.substring(0, minMatchSize); 
101                      }else{ 
102                          myMiniumWord = wordOfBD; 
103                      } 
104                      // Si la palabra del diccionario es diferente de nulo y cadena vacia 
105                      if(myMiniumWord != null && myMiniumWord != ""){ 
106                          // Se obtiene un patron de caracteres de las palabras del diccionario (Con base en consonantes) 
107                          String patronConsonante = myMiniumWord.replaceAll("(?![aeiouAEIOU])[a-zA-Z]","[bcdefghijklmnopqrstuvwxyzBCDEFGHIJKLMNOPQRSTUVWXYZ]+"); 
108                          // Se obtiene un patron de caracteres de las palabras del diccionario (Con base en vocales) 
109                          String patronVocal = myMiniumWord.replaceAll("(?![bcdefghijklmnopqrstuvwxyzBCDEFGHIJKLMNOPQRSTUVWXYZ])[a-zA-Z]", "[aeiouAEIOU]+"); 
110                          Log.d(TAG, "Patron.Constante:_"+patronConsonante+"______"); 
111                          Log.d(TAG, "Patron.Vocal:_____"+patronVocal+"______"); 
112                          Matcher matCon = Pattern.compile(patronConsonante).matcher(wordString); // Comparar la palabra entrante contra el patron de la palabra del diccionario (con base en consonantes) 
113                          Matcher matVoc = Pattern.compile(patronVocal).matcher(wordString);      // Comparar la palabra entrante contra el patron de la palabra del diccionario (con base en vocales) 
114                          // Si algun patron coincide entonces se toma a la palabra del diccionario, como la palabra que se esta escribiendo 
115                          if (matCon.matches() || matVoc.matches()) { 
116                              newWord = wordOfBD + " "; // Establecer la nueva palabra 
117                              breakFixed = true;        // Indicar la salida del ciclo tras predecir la palabra 
118                              break; 
119                          } 
120                      } 
121                  } 
122                  if(breakFixed){ 
123                      break; 
124                  } 
125              } 
126          } 
127          return newWord; 
128      }
