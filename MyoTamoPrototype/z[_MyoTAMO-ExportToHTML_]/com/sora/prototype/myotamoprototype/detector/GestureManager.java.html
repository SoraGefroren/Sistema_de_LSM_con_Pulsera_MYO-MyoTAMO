306       * @param emgDataDetected Dato EMG 
307       * @return Indicador de que han sido o no juntadas una determinada cantidad de muestras 
308       */ 
309      private boolean goHandGestureManager(EmgData_Number emgDataDetected){ 
310          if(!busyArrayAccumulated){ // Pregunta si estan ocupando el almacen de muestras 
311              // ======================= =========================== 
312              GestureManager.padlock_arrayEmgDataAccumulated.lock(); // Cerrar candado de almacen de muestras 
313              if( arrayEmgDataNumberAccumulated.size() < MAXSizeArray ){  // Si no se ah alcanzado el maximo de muestras 
314                  arrayEmgDataNumberAccumulated.add(emgDataDetected);     // Guardar datos EMG en el almacen de muestras 
315                  sizeArrayAccumulated = arrayEmgDataNumberAccumulated.size(); // Actualizarl el numero de muestras 
316              } 
317              GestureManager.padlock_arrayEmgDataAccumulated.unlock(); // Liberar candado de almacen de muestras 
318              // ======================= ========================== 
319              if(arrayEmgDataNumberAccumulated.size() >= MAXSizeArray){ // Si se ha alcanzdo el maximo de muestras 
320                  return true; 
321              } 
322          } 
323          return false; 
324      }
